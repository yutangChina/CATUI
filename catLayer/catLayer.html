function openLayer() {
  var catLayer = $.catLayer({
    content: '<div style="width:500px;"><a class="gy-btn blueBtn fl" style="margin-left: 0px;" id="closeBtn">关闭弹框</a></div>',
  });
  $(catLayer.getContent()).on('click', '#closeBtn', function () {
     catLayer.closeLayer();
});
